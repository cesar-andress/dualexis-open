\subsection{Work Packages}

\begin{enumerate}[noitemsep]
  \item \textbf{WP-GOV} --- governance, DPIA, ethics, transparency materials.
  \item \textbf{WP-NET} --- trust-zone segmentation, data-flow sign-off, encrypted staff access
        (engineering detail: Appendix~\ref{app:technical-architecture}).
  \item \textbf{WP-HW} --- edge nodes, orchestrator, UPS, mounting.
  \item \textbf{WP-SW} --- DUALEXIS deployment, dashboard integration, monitoring.
  \item \textbf{WP-OPS} --- SOPs, training, tabletop exercises.
  \item \textbf{WP-EVAL} --- metrics collection, reporting, phase reviews.
\end{enumerate}

\subsection{Deliverables by Phase}

\paragraph{Phase 0--1.}
Governance charter; network diagram; sandbox acceptance test; training materials.

\paragraph{Phase 2 (MVP).}
Live pilot report; false-alarm log; DPIA update; operational metrics dashboard.

\paragraph{Phase 3--4.}
Multi-site harmonization guide; federated metrics specification; peer-reviewed methods paper (optional).

\subsection{Quality Assurance}

\begin{itemize}[noitemsep]
  \item acceptance tests: no raw media on bus; biometric fields rejected;
  \item penetration test of dashboard VLAN (recommended before extended pilot);
  \item quarterly policy compliance audit during active sensing;
  \item version-controlled configuration manifests per site.
\end{itemize}

\subsection{Dependencies and Assumptions}

Assumptions:
\begin{itemize}[noitemsep]
  \item school provides VLAN capacity and physical access for edge installs;
  \item safety officer capacity for review during staffed hours;
  \item DPO availability for DPIA within institutional timelines;
  \item camera infrastructure exists or is procured separately from DUALEXIS software scope.
\end{itemize}

Dependencies:
\begin{itemize}[noitemsep]
  \item ethics approval before human-subjects surveys;
  \item signed DPAs before research team accesses operational logs;
  \item Horizon funding (if applicable) for extended multi-site phase only.
\end{itemize}
